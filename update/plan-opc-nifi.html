mintpower: opc_server.py ──── Python asyncua (simulate ในตัว) ──── Kafka: opc-raw-data ↑ ส่ง JSON ตรง via NodePort 32092 NiFi Edge ── ConsumeKafka ──→ PublishKafka(opc-metrics) └──→ PublishKafka(opc-datalake) ⚠ NiFi Edge ไม่ได้อ่าน OPC UA จริง เป็นแค่ Kafka fan-out ⚠ mintserver:53530 ไม่ได้ถูกใช้งาน

// ============================================================
// OPC UA Reader — Eclipse Milo + NiFi ExecuteGroovyScript
// อ่านข้อมูล Prosys บน mintserver ส่งเป็น JSON FlowFile
// ============================================================

import org.eclipse.milo.opcua.sdk.client.OpcUaClient
import org.eclipse.milo.opcua.sdk.client.api.config.OpcUaClientConfig
import org.eclipse.milo.opcua.stack.client.DiscoveryClient
import org.eclipse.milo.opcua.stack.core.types.builtin.NodeId
import org.eclipse.milo.opcua.stack.core.types.builtin.LocalizedText
import org.eclipse.milo.opcua.stack.core.types.enumerated.TimestampsToReturn
import org.apache.nifi.processor.io.OutputStreamCallback
import groovy.json.JsonOutput

// ── Config ──────────────────────────────────────────
@groovy.transform.Field
static final String OPC_ENDPOINT =
    "opc.tcp://10.85.3.100:53530/OPCUA/SimulationServer"

@groovy.transform.Field
static final String SOURCE_ID = "mintserver-prosys"

@groovy.transform.Field
static final String DEVICE_ID = "opc-prosys-300tags"

// ── Node IDs (แก้ตามค่าจริงจาก Prosys) ─────────────
// Prosys default 7 tags (ns=3)
@groovy.transform.Field
static final Map<String, NodeId> NODES = [
    "Counter"  : new NodeId(3, 1001),
    "Random"   : new NodeId(3, 1002),
    "Sawtooth" : new NodeId(3, 1003),
    "Sinusoid" : new NodeId(3, 1004),
    "Square"   : new NodeId(3, 1005),
    "Triangle" : new NodeId(3, 1006),
    "Constant" : new NodeId(3, 1007),
    // Custom 300 tags (เพิ่มหลัง browse NodeID จริง)
    // "Temp_Boiler_01": new NodeId(3, 2001),
    // "Temp_Boiler_02": new NodeId(3, 2002),
    // ... ครบ 300 ...
]

// ── Static OPC Client (ไม่ reconnect ทุก 2 วิ) ───────
@groovy.transform.Field
static OpcUaClient opcClient = null

@groovy.transform.Field
static final Object LOCK = new Object()

def ensureConnected() {
    synchronized (LOCK) {
        if (opcClient == null) {
            log.info("Connecting to OPC UA: ${OPC_ENDPOINT}")
            def endpoints = DiscoveryClient
                .getEndpoints(OPC_ENDPOINT).get()
            def config = OpcUaClientConfig.builder()
                .setEndpoint(endpoints[0])
                .setApplicationName(
                    LocalizedText.english("NiFi OPC Client"))
                .setApplicationUri("urn:nifi:opc:client")
                .build()
            opcClient = OpcUaClient.create(config)
            opcClient.connect().get()
            log.info("OPC UA connected ✓")
        }
    }
}

// ── Main ──────────────────────────────────────────
try {
    ensureConnected()

    // Batch read (เร็วกว่าอ่านทีละตัว)
    def nodeList = NODES.values().toList()
    def values   = opcClient
        .readValues(0.0, TimestampsToReturn.Both, nodeList).get()

    def tags = [:]
    NODES.eachWithIndex { entry, i ->
        def dv = values[i]
        tags[entry.key] = dv?.statusCode?.isGood()
            ? dv.getValue()?.getValue()
            : null
    }

    def payload = [
        timestamp : System.currentTimeMillis() / 1000.0,
        source_id : SOURCE_ID,
        device_id : DEVICE_ID,
        tags      : tags
    ]

    def ff = session.create()
    ff = session.write(ff, { out ->
        out.write(JsonOutput.toJson(payload).getBytes("UTF-8"))
    } as OutputStreamCallback)
    ff = session.putAttribute(ff, "mime.type", "application/json")
    ff = session.putAttribute(ff, "tag.count", "${tags.size()}")
    session.transfer(ff, REL_SUCCESS)

} catch (Exception e) {
    log.error("OPC read error: ${e.message}", e)
    synchronized (LOCK) { opcClient = null }   // reset → reconnect
    def ff = session.create()
    ff = session.putAttribute(ff, "error.message", e.message ?: "unknown")
    session.transfer(ff, REL_FAILURE)
}

Browse Node IDs จาก Prosys

ต้อง browse หา Node ID จริงก่อนใส่ใน Groovy script เพราะ Prosys อาจ assign ID ต่างจากที่กำหนดใน XML

# /home/mintpower/lab/k3s/tools/browse_opc.py
import asyncio
from asyncua import Client

async def browse():
    async with Client("opc.tcp://10.85.3.100:53530/OPCUA/SimulationServer") as c:
        # Browse namespace 3 (simulation tags)
        ns = await c.get_namespace_index("urn:prosys:ua:simserver")
        root = c.get_node("ns=0;i=84")

        async def walk(node, depth=0):
            try:
                name = await node.read_browse_name()
                nid  = node.nodeid
                if nid.NamespaceIndex == 3:
                    print(f"  ns={nid.NamespaceIndex};i={nid.Identifier}  →  {name.Name}")
                for child in await node.get_children():
                    await walk(child, depth+1)
            except: pass

        await walk(root)

asyncio.run(browse())

ปัญหาที่อาจเจอและวิธีแก้

อาการสาเหตุวิธีแก้
ClassNotFoundException ใน NiFi log JAR ขาด หรือ Module Directory path ผิด ตรวจ path /opt/nifi/nifi-current/data/milo-jars/ มีไฟล์ครบไหม, ตรวจ property Module Directory ใน processor
Connect timeout ไป mintserver Prosys ไม่รัน หรือ firewall กัน port 53530 nc -zv 10.85.3.100 53530 จาก NiFi pod: kubectl exec -n it $POD -- nc -zv 10.85.3.100 53530
Node ID ไม่เจอ / Bad Quality NodeId ใน script ไม่ตรงกับ Prosys รัน browse_opc.py แล้ว update NODES map ใน script
NiFi processor error ทุก 2 วิ Script error / import ผิด ดู NiFi Bulletin Board (icon ระฆัง) → copy error → แก้ script
Kafka ไม่มีข้อมูลใหม่ PublishKafka processor ยังไม่ start Start processor ใน NiFi UI, ตรวจ Connection ระหว่าง processor
Groovy script import error (Milo version) JAR เป็น version อื่นที่ API ต่างออกไป ตรวจให้แน่ใจว่าใช้ 0.6.12 ครบทุกไฟล์, ไม่ใช้ 0.7.x ซึ่ง API เปลี่ยน
ข้อมูลซ้ำ (duplicate) ใน Kafka opc_server.py ยังรันอยู่ sudo systemctl stop opc-simulator
